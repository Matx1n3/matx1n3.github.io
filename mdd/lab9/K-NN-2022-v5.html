ctrl <- trainControl("cv", number=10)

# ¿Cómo se codifica cuáles son las predictoras, y cuál la clase a predecir?
# Al menos, describe los siguientes parámetros: "method", "tuneLength", "trControl", "preProc"
# ¿Ves algún otro parámetro interesante en la ayuda de la función "train? ¿Cuáles?
KNNModel1 <- train(variety ~ ., data=trainSet, method="knn", tuneLength=5, trControl=ctrl, preProc=c("scale"))
# El primer parametro de la funcion indica cuales son las variables predictoras y cual es la clase a predecir
# method indica el metodo de aprendizaje automatico a utilizar
# tuneLength indica la cantidad de configuraciones de hyperparametros a probar
# trControl usa el objeto ctrl previamente instanciado para controlar el entrenamiento. En este caso se usara cross-validation con 10 folds
# preProc indica que preprocesamiento se empleara. En este caso se escalan las variables predictoras
